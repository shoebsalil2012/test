	function filtername(prodname) {

		if (prodname.length >= minlength ) {

							if(currentRequest != null) {
								currentRequest.abort();
							}
							},
							success: function(result){
								$('#productlist').replaceWith(result);
								$('#loader-box').hide();
							}});
		}
		
	}
